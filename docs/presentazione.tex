% ---------------- Il problema / la soluzione ----------------
\begin{frame}{L'idea}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Il problema}
        Le informazioni di un viaggio --- itinerario, prenotazioni, valigia, spese --- sono
        \textbf{sparse} tra più app e spesso richiedono la \textbf{rete}.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{La soluzione}
        Un'unica app che raccoglie tutto e funziona \textbf{100\% offline}:
        i dati sono sempre a portata di mano.
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.7cm}
  \begin{center}
    \large\bfseries
    \textcolor{primary}{Pianifica} \quad\textbullet\quad
    \textcolor{primary}{Gestisci} \quad\textbullet\quad
    \textcolor{primary}{Analizza}
  \end{center}
\end{frame}

% ---------------- Feature avanzate ----------------
\begin{frame}{Feature avanzate}
  \begin{block}{1 --- Duplicazione completa del viaggio}
    Copia un viaggio con \textbf{tappe, attività e checklist}, ricollegando
    correttamente ogni elemento alle nuove tappe. Ripianificare un viaggio simile è
    questione di un tocco.
  \end{block}
  \begin{block}{2 --- Packing list dal tipo di viaggio}
    La valigia si genera in automatico in base al \textbf{tipo di viaggio} scelto
    (mare, montagna, città, business, avventura, inverno): oggetti di base + specifici.
  \end{block}
  \begin{center}\small\color{soft}
    In più: \textbf{timeline} cronologica di tappe e attività.
  \end{center}
\end{frame}

% ---------------- Scelte progettuali ----------------
\begin{frame}{Scelte progettuali}
  \begin{itemize}
    \item \textbf{Flutter} --- unica base di codice, UI Material 3 coerente.
    \item \textbf{Provider} --- gestione dello stato semplice e reattiva.
    \item \textbf{SQLite} --- dati relazionali con integrità referenziale, tutto in locale.
    \item \textbf{Stato del viaggio calcolato} dalle date: l'elenco resta sempre coerente
      nel tempo, senza aggiornamenti manuali.
  \end{itemize}
\end{frame}

% ---------------- Difficoltà ----------------
\begin{frame}{Difficoltà incontrate}
  \begin{block}{Coerenza dei riferimenti nella duplicazione}
    Duplicando le tappe cambiano gli identificatori: attività e checklist rischiano di
    puntare a tappe inesistenti. \textbf{Soluzione:} una mappa
    \textit{id originale $\rightarrow$ id copia} per ricollegare tutto correttamente.
  \end{block}
  \begin{block}{Integrità dei dati offline}
    Eliminazioni a cascata e stato calcolato garantiscono dati sempre coerenti anche
    senza rete.
  \end{block}
\end{frame}
